        {
            try
            {
                var response = {{variable.service}}.GetById({{parameters.init}});

            }
            catch (Exception ex)
            {
                throw ex;
            }
        }
        [Route("GetById")]
        [HttpGet]
        [AllowAnonymous]
        public IHttpActionResult GetById ([FromUri]string id)
        {
            try
            {
                var responseData = _postCategoryService.GetSingleById(id);
                return Ok(responseData);
            }
            catch (Exception ex)
            {
                _logger.LogError("Error at method: GetSingle - PostCategoryApi," + ex.Message + "");
                return BadRequest("Error System");
            }
        }
