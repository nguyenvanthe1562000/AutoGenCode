CREATE PROCEDURE [dbo].[sp_{{nametable}}_delete_multi]
(
  {{store.parameter_1}}
)
AS
    BEGIN
        DECLARE @str_id NVARCHAR(50);
        SELECT {{store.selectjsonfield}}
        INTO #Temp
        FROM {{store.fromjson}};
        WHILE
        (
            SELECT COUNT(*)
            FROM #Temp
        ) > 0
            BEGIN
                SELECT TOP 1 @str_id = employee_id
                FROM #Temp;
               DELETE FROM {{nametable}} WHERE {{store.required_3}}
                
                DELETE #Temp
                WHERE {{store.required_3}}
            END;
        DROP TABLE #Temp;
    END;
GO;
