[Route("GetById/{{parameters.route}}")]
[ClaimRequirement(ClaimFunction.FUNCTION, ClaimAction.CANREAD)]
 public async Task<IActionResult> GetById ({{parameters.fromuri}})
{
    try
    {
        var responseData =await {{variable.service}}.GetById({{parameters.pass}});

    }

    }
}
