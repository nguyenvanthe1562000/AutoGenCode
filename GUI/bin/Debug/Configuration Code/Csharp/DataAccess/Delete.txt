
public async Task<bool>  Delete({{parameters.init}}, int userId)
{

        try
        {
                var result =await _dbHelper.ExecuteScalarSProcedureWithTransactionAsync( "{{nametable}}_delete",{{parameters.store}},"@user_id", userId);
                if (!string.IsNullOrEmpty(result.message.ToString()))
                {
                    return false;
                    throw new Exception(result.message);
                }
                 return true;
            }
            catch (Exception ex)
            {
                throw ex;
            }
}
